
% Century Schoolbook-like font
\usepackage{newcent}

% Colors and page background
\usepackage{xcolor}
\definecolor{forrtbg}{HTML}{FEFDF6}
\definecolor{FORRTblue}{HTML}{2C3E50}
\pagecolor{forrtbg}

% Section heading colors
\addtokomafont{section}{\color{FORRTblue}}
\addtokomafont{subsection}{\color{FORRTblue}}

% Safe link color configuration
\AtBeginDocument{
  \hypersetup{
    colorlinks=true,
    linkcolor=FORRTblue,
    urlcolor=FORRTblue,
    citecolor=FORRTblue
  }
}

% Line spacing
\usepackage{setspace}
\setstretch{1.5}

% Smaller tables
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\small}
\AtBeginEnvironment{tabular}{\small}

% Avoid auto title page
\makeatletter
\renewcommand{\maketitle}{}%
\makeatother

% Prevent forced blank recto pages
\makeatletter
\let\cleardoublepage\clearpage
\makeatother

% --- TOC layout & spacing ---
% Let KOMA handle TOC layout and ensure wrapping is sane
\KOMAoptions{toc=graduated}

% Keep TOC single-spaced only inside TOC
\BeforeTOCHead[toc]{\begingroup\setstretch{1}\selectfont}
\AfterTOCHead[toc]{\endgroup}

% Optional: do the same for LoF/LoT
\BeforeTOCHead[lof]{\begingroup\setstretch{1}\selectfont}
\AfterTOCHead[lof]{\endgroup}
\BeforeTOCHead[lot]{\begingroup\setstretch{1}\selectfont}
\AfterTOCHead[lot]{\endgroup}

% TOC depth and number column width for nicer wrapping/alignment
\setcounter{tocdepth}{2}
\RedeclareSectionCommand[tocnumwidth=3.5em]{chapter}

% Make pages stretch to avoid bottom clipping
\flushbottom
