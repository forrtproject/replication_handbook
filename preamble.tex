% Font
\usepackage{newcent}

% Colour + background (optional; safe)
\usepackage{xcolor}
\definecolor{forrtbg}{HTML}{FEFDF6}
\definecolor{FORRTblue}{HTML}{2C3E50}
\pagecolor{forrtbg}

% Headings colour (optional)
\addtokomafont{section}{\color{FORRTblue}}
\addtokomafont{subsection}{\color{FORRTblue}}

% Safe link colours
\AtBeginDocument{
  \hypersetup{
    colorlinks=true,
    linkcolor=FORRTblue,
    urlcolor=FORRTblue,
    citecolor=FORRTblue
  }
}

% Line spacing
\usepackage{setspace}
\setstretch{1.5}

% Smaller tables
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\small}
\AtBeginEnvironment{tabular}{\small}

% Do NOT generate an auto title page
\makeatletter
\renewcommand{\maketitle}{}%
\makeatother

% Avoid forced blank recto pages
\makeatletter
\let\cleardoublepage\clearpage
\makeatother

% TOC layout (wraps nicely)
\KOMAoptions{toc=graduated}

% Let pages stretch so bottoms align (prevents “clipping” feel)
\flushbottom
